561 void ListeLecture::SuppressionLigne()
562 {
563  m_supprEnCours = true;
564  if (GetItemCount() == 0)
565  return;
566 
567  #if DEBUG
568  FichierLog::Get()->Ajouter(_T("ListeLecture::SuppressionLigne - Début"));
569  #endif
570  long position = GetNextItem(-1, wxLIST_NEXT_ALL, wxLIST_STATE_SELECTED);
571  if (position != -1 && GetItemCount() > 0)
572  {
573  if (GetSelectedItemCount() == 1)
574  {
575  #if DEBUG
576  FichierLog::Get()->Ajouter(_T("ListeLecture::SuppressionLigne - 1 ligne"));
577  #endif
578  DeleteItem(position);
579  for (size_t k = 0; k < m_ocurrenceLigne.GetCount(); k++)
580  {
581  if (m_ocurrenceLigne.Item(k) >= position)
582  m_ocurrenceLigne.Item(k)--;
583  }
584  m_musique->SupprimerNom(position);
585  }
586  else
587  {
588  long i = 0, j = 0, max = GetSelectedItemCount();
589 
590  #if DEBUG
591  FichierLog::Get()->Ajouter(_T("ListeLecture::SuppressionLigne - ") + wxString::Format(_T("%ld lignes"), max));
592  #endif
593 
594  wxProgressDialog barProgre(_("Mise à jour"), _("Suppression en cours"), max);//
595  wxString chemin;
596  wxArrayString tableau;
597  wxListItem item;
598 
599  tableau.Alloc(max);
600  while (i < max)
601  {
602  j = GetNextItem(-1, wxLIST_NEXT_ALL, wxLIST_STATE_SELECTED);
603  item.SetId(j);
604  item.SetColumn(6);
605  item.SetMask(wxLIST_MASK_TEXT);
606  GetItem(item);
607 
608  chemin = item.GetText();
609  item.SetColumn(0);
610  GetItem(item);
611  chemin << wxFileName::GetPathSeparator() << item.GetText();
612 
613  tableau.Add(chemin);//
614  DeleteItem(j);
615  i++;
616  barProgre.Update(i);//Pulse();//
617 
618  for (size_t k = 0; k < m_ocurrenceLigne.GetCount(); k++)
619  {
620  if (m_ocurrenceLigne.Item(k) >= j)
621  m_ocurrenceLigne.Item(k)--;
622  }
623  }
624  m_musique->SupprimerNom(&tableau);
625  tableau.Clear();
626  }
627 
628  if (GetItemCount() == 0)
629  {
630  wxCommandEvent evt(wxEVT_VIDER_PANNEAU, GetId());
631  GetParent()->GetEventHandler()->AddPendingEvent(evt);
632  }
633  else if (GetItemCount() > position)//Toujours avoir une ligne active
634  SetItemState(position, wxLIST_STATE_FOCUSED|wxLIST_STATE_SELECTED, wxLIST_STATE_FOCUSED|wxLIST_STATE_SELECTED);
635  else if (GetItemCount() >= 1)
636  SetItemState(position-1, wxLIST_STATE_FOCUSED|wxLIST_STATE_SELECTED, wxLIST_STATE_FOCUSED|wxLIST_STATE_SELECTED);
637  GestPeriph::Get()->MAJPlaylist();
638  SetFocus();
