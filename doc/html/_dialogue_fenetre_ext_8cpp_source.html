97 wxString DialogEnregistreM3U::GetCheminRaccourci()
98 {
99  if (m_boiteRaccourci->IsEmpty() || m_boiteRaccourci->GetValue().StartsWith(m_rep))
100  return wxEmptyString;
101  else if (!m_boiteNouveau->IsEmpty())
102  return m_boiteRaccourci->GetValue() + wxFileName::GetPathSeparator() + m_boiteNouveau->GetValue() + _(" (raccourci).lnk");
103  else if (m_liste->GetCurrentSelection() != wxNOT_FOUND)
104  return m_boiteRaccourci->GetValue() + wxFileName::GetPathSeparator() + m_chaineM3U.Item(m_liste->GetCurrentSelection()).AfterLast(wxFileName::GetPathSeparator()).BeforeLast('.') + _(" (raccourci).lnk");
105  else
106  return wxEmptyString;
107 }
