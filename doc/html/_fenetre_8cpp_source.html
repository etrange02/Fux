71 FuXFenetre::FuXFenetre(int argc, wxChar **argv) : wxFrame(NULL, wxID_ANY, _T("Fu(X) 2.0"))
72 {
73  wxMutexLocker lock(s_mutexProtectionDemarrage);
74  m_FenetreActuel = PRINCIPAL;
75  m_nouvelleFenetre = PRINCIPAL;
76  sizerPrincipalH = new wxBoxSizer(wxHORIZONTAL);
77  sizerGaucheV = new wxBoxSizer(wxVERTICAL);
78  sizerPrincipalH->Add(sizerGaucheV, 0, wxALL | wxEXPAND, 0);
79  m_MAJliste = false;
80 
81  ConstructionBarre();
82  ConstructionSizerGauche();
83 
84  Initialisation();
85  CreerPages();
86 
87  SetSizer(sizerPrincipalH);
88  sizerPrincipalH->SetSizeHints(this);
89 
90  LecturePreference(argc <= 1);
91 
92  if (argc > 1)
93  {
94  wxFileName fichierMem(argv[1]);
95  fichierMem.Normalize(wxPATH_NORM_SHORTCUT);
96  if (fichierMem.GetExt().Lower() == _T("m3u"))
97  {
98  wxTextFile test(fichierMem.GetFullPath()); test.Open();
99  if (test.IsOpened())
100  {
101  if (test.GetLineCount() > 1)
102  {
103  m_chanson->CopieFichier(fichierMem.GetFullPath());
104  m_chanson->ChangementChanson();
105  m_MAJliste = true;
106  }
107  else wxLogMessage(_("Impossible de charger le fichier, celui-ci est vierge !"));
108  test.Close();
109  }
110  }
111  else if (Parametre::Get()->islisable(fichierMem.GetExt().Lower()))
112  {
113  m_chanson->Lecture(fichierMem.GetFullPath());
114  m_chanson->Listage();
115  m_MAJliste = true;
116  }
117  }
118 
119  SetIcon(wxIcon(Parametre::Get()->getRepertoireExecutableLib(_T("play.ico")), wxBITMAP_TYPE_ICO));
120 
121  m_TimerGraph.setMusique();
122  if (m_MAJliste)
123  m_playList->GetListeLecture()->MAJ();
124 
125  ChangeFenetre();
126  m_serveur = new TCPServeur(this);
127  if (!m_serveur->Create(IPC_SERVICE))
128  {
129  wxLogMessage(_T("Impossible d'initialiser le serveur"));
130  }
131 
132  m_fenetresDetachables = new ArrayFenetreDetachable();
133  #if DEBUG
134  FichierLog::Get()->Ajouter(_T("Fin de FuXFenetre::FuXFenetre(int argc, wxChar **argv)"));
135  #endif
136 }
