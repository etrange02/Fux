62 void CouperCopierColler::ModifEtatDonnee(bool reset)
63 {
64  if (m_action == COUPER)
65  {
66  if (m_objetOrigine == M3U)
67  {
68  wxTextFile fichier(m_cheminOri);
69  if (!fichier.Open())
70  return;
71 
72  size_t max = m_tableau.GetCount(), i = 0, j = 0;
73  bool parcouru = true, cont = true;
74 
75  while (i<max && cont)
76  {
77  while (j<fichier.GetLineCount() && cont)
78  {
79  if (fichier.GetLine(j).IsSameAs(m_tableau.Item(i)))
80  {
81  fichier.RemoveLine(j);
82  i++;
83  j--;
84  parcouru = false;
85  if (i >= max)
86  cont = false;
87  }
88  j++;
89  }
90  if (parcouru)
91  { i++; parcouru = false;}
92  else
93  parcouru = true;
94  j = 0;
95  }
96 
97  fichier.Write();
98  fichier.Close();
99  reset = true;
100  }
101  else if (m_objetOrigine == PLAYLIST)
102  {
103  Musique::Get()->SupprimerNom(&m_tableau, true);
104  reset = true;
105  }
106  }
107  if (reset)
108  {
109  m_tableau.Empty();
110  m_action = -1;
111  m_objetOrigine = VIDE;
112  }
113 }

121 bool CouperCopierColler::SetDestination(wxString destination, int type)
122 {
123  bool fait = false;
124  if (type == VIDE)
125  return fait;
126  if (type == DOSSIER && m_objetOrigine == DOSSIER)
127  {
128  wxArrayString *tableau = new wxArrayString;
129  *tableau = m_tableau;
130  ThreadFichierFichier::Get()->AjoutDonnee(tableau, (m_action == COPIE) ? COPIE : DEPLACE, destination);
131  if (!ThreadFichierFichier::Get()->GetEtat())
132  ThreadFichierFichier::Get()->Run();
133  if (m_action == COUPER)
134  ModifEtatDonnee(true);
135  fait = true;
136  }
137  else if (type == M3U)
138  {
139  if (!wxFileExists(destination))
140  return false;
141 
142  wxTextFile fichier(destination);
143  if (!fichier.Open())
144  return false;
145 
146  size_t max = m_tableau.GetCount();
147  for (size_t i=0; i<max; i++)
148  {
149  if (wxFileExists(m_tableau.Item(i)) && Parametre::Get()->islisable(m_tableau.Item(i).AfterLast('.').Lower()))
150  fichier.AddLine(m_tableau.Item(i));
151  else
152  wxLogMessage(_("Fichier introuvable : ") + m_tableau.Item(i));
153  }
154  fichier.Write();
155  fichier.Close();
156  fait = true;
157  ModifEtatDonnee();
158  }
159  else if (type == PLAYLIST)
160  {
161  size_t i = 0;
162  while(i < m_tableau.GetCount())
163  {
164  if (Parametre::Get()->islisable(m_tableau.Item(i).AfterLast('.').Lower()))
165  i++;
166  else
167  m_tableau.RemoveAt(i);
168  }
169  if (m_tableau.IsEmpty())
170  return false;
171 
172  Musique::Get()->Listage(&m_tableau, true);
173  fait = true;
174  ModifEtatDonnee();
175  }
176 
177  return fait;
178 }
