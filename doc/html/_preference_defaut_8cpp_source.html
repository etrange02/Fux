158 int PrefDefaut::CreerListe(wxChoice *liste)
159 {
160  wxString chemin = Parametre::Get()->getRepertoireParametre(_T("Liste_M3U.txt"));
161 
162  wxString copie;
163  wxDir Repertoire(Parametre::Get()->getRepertoireParametre(_T("Play_list_M3U")));
164  wxFile fichier;
165  fichier.Create(chemin, true);
166  int nombre = 0;
167  bool continuer = Repertoire.GetFirst(&copie, _T("*.m3u"));//, wxDIR_FILES);
168 
169  while (continuer)
170  {
171  fichier.Write(copie);
172  fichier.Write(_T("\r\n"));
173  nombre++;
174  liste->Append(copie);
175  continuer = Repertoire.GetNext(&copie);
176  }
177  fichier.Close();
178  return nombre;
179 }

187 int PrefDefaut::ListerContenuFichier(wxChoice *liste, int typeListe)

189  wxString chemin = _T(""), chaine;

191 
192  if (typeListe == COULEUR)

197  wxTextFile fichier;

205  }
206 
207  fichier.Close();

214 void PrefDefaut::Defaut_Bouton_Enregistrer(wxCommandEvent &WXUNUSED(event))
215 {

219  if (m_checkBox[COULEUR].GetValue() && m_listeCouleur->GetCurrentSelection() != wxNOT_FOUND)

233  else if (m_listeM3u_mp3 == M3U && !m_listeReprise->GetCurrentSelection() != wxNOT_FOUND)
234  chaineReprise << _T("Reprise= M3U\r\nChemin= ") << m_listeReprise->GetString(m_listeReprise->GetCurrentSelection());
235  else
236  chaineReprise << _T("Reprise= NON\r\nChemin= NON");
237  }
238  else
239  chaineReprise << _T("Reprise= NON\r\nChemin= NON");
240 
241  if (m_checkBox[SOUSDOSSIER].GetValue())
242  {
243  chaineRecherche << _T("SousDossier= OUI");
244  Parametre::Get()->setSousDossier(true);
245  }
246  else
247  {
248  chaineRecherche << _T("SousDossier= NON");
249  Parametre::Get()->setSousDossier(false);
250  }
251 
252  FichierListe.Create(chemin, true);
253  FichierListe.Write(_T("#EXTCONF_1\r\n"));
254  FichierListe.Write(chaineCouleur + _T("\r\n"));
255  FichierListe.Write(chaineSon + _T("\r\n"));
256  FichierListe.Write(chaineReprise + _T("\r\n"));
257  FichierListe.Write(chaineRecherche + _T("\r\n"));
258  FichierListe.Close();
259 }

431 void PrefDefaut::OuvertureFichier()
432 {
433  wxString cheminFichier = Parametre::Get()->getRepertoireParametre(_T("Fu(X).conf")), nomFichierCouleur, nomFichierSon;
434  wxTextFile fichierPref(cheminFichier);
435 
436  if (!(fichierPref.Exists() && fichierPref.Open()))
437  return;
438 
439  if (fichierPref.GetLine(0).IsSameAs(_T("#EXTCONF_1")))
440  {
442  if (!fichierPref.GetLine(1).IsSameAs(_T("Couleur= NON")))
443  {
444  m_checkBox[COULEUR].SetValue(true);
445  m_listeCouleur->Enable(true);
446  m_listeCouleur->SetStringSelection(fichierPref.GetLine(1).AfterFirst(' '));
447  }
449  if (!fichierPref.GetLine(2).IsSameAs(_T("Son= NON")))
450  {
451  m_checkBox[SON].SetValue(true);
452  m_listeSon->Enable(true);
453  m_listeSon->SetStringSelection(fichierPref.GetLine(2).AfterFirst(' '));
454  }
456  if (!fichierPref.GetLine(3).IsSameAs(_T("Reprise= NON")))
457  {
458  m_checkBox[REPRISE].SetValue(true);
459  m_choix->Enable(true);
460  if (fichierPref.GetLine(3).IsSameAs(_T("Reprise= M3U")))
461  {
462  m_listeReprise->Enable(true);
463  m_listeReprise->SetStringSelection(fichierPref.GetLine(4).AfterFirst(' '));
464  m_choix->SetSelection(M3U);
465  }
466  else if (fichierPref.GetLine(3).IsSameAs(_T("Reprise= MP3")))
467  {
468  m_boiteCheminChansonRep->Enable(true);
469  m_boutonRecherche->Enable(true);
470  m_boiteCheminChansonRep->ChangeValue(fichierPref.GetLine(4).AfterFirst(' '));
471  m_choix->SetSelection(MP3);
472  }
473  }
475  if (!fichierPref.GetLine(5).IsSameAs(_T("SousDossier= NON")))
476  m_checkBox[SOUSDOSSIER].SetValue(true);
477  }
478 
479  fichierPref.Close();
480 }
