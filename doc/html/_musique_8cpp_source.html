587 void Musique::SupprimerNom(wxArrayString *tableau, bool maj)
588 {
589  unsigned int i = 0, taille = tableau->GetCount(), pos = 0;
590  m_positionListe = m_liste->GetPositionListe(m_cheminComplet, m_positionListe);
591  bool change = false;
592 
593  while (i<taille && change == false)
594  {
595  if (m_cheminComplet.IsSameAs(tableau->Item(i)))
596  {
597  change = true;
598  pos = m_positionListe - i;
599  }
600  i++;
601  }
602 
603  m_liste->EffacerNom(tableau);
604 
605  if (change)
606  {
607  if (m_liste->GetNombreFichier() > 1)
608  {
609  if (m_positionListe >= m_liste->GetNombreFichier())
610  m_positionListe = 0;
611  else
612  m_positionListe = pos;
613  ChangementChanson(m_positionListe, m_liste->GetNomPosition(m_positionListe));
614  }
615  else
616  ModifListeVide();
617  }
618  Recharger(maj, false);
619 }
