275 void Parametre::creerRepertoireParametre(wxString chemin)
276 {
277  wxString repertoire;
278 
279  //Test répertoire générale
280  repertoire = chemin;
281  if(!wxFileName::DirExists(repertoire))
282  wxFileName::Mkdir(repertoire);
283 
284  //Test sous-Rep Pref
285  repertoire = chemin + wxFileName::GetPathSeparator() + _T("Preference");
286  if(!wxFileName::DirExists(repertoire))
287  wxFileName::Mkdir(repertoire);
288 
289  //Test sous-Rep Pref/couleur
290  repertoire = chemin + wxFileName::GetPathSeparator() + _T("Preference") + wxFileName::GetPathSeparator() + _T("Couleur");
291  if(!wxFileName::DirExists(repertoire))
292  wxFileName::Mkdir(repertoire);
293 
294  //Test sous-Rep Pref/son
295  repertoire = chemin + wxFileName::GetPathSeparator() + _T("Preference") + wxFileName::GetPathSeparator() + _T("Son");
296  if(!wxFileName::DirExists(repertoire))
297  wxFileName::Mkdir(repertoire);
298 
299  //Test sous-Rep Playlist
300  repertoire = chemin + wxFileName::GetPathSeparator() + _T("Play_list_M3U");
301  if(!wxFileName::DirExists(repertoire))
302  wxFileName::Mkdir(repertoire);
303 
304  //Test fichier conf
305  repertoire = chemin + wxFileName::GetPathSeparator() + _T("Fu(X).conf");
306  if(!wxFileName::FileExists(repertoire))
307  {
308  wxTextFile fichier(repertoire);
309  if (fichier.Create())
310  {
311  fichier.AddLine(_T("#EXTCONF_1"));
312  fichier.AddLine(_T("Couleur= NON"));
313  fichier.AddLine(_T("Son= NON"));
314  fichier.AddLine(_T("Reprise= NON"));
315  fichier.AddLine(_T("Chemin= NON"));
316  fichier.AddLine(_T("SousDossier= NON"));
317 
318  fichier.Write();
319  fichier.Close();
320  }
321  }
322 }
