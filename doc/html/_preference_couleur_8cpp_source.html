280 int PrefCouleur::CreerListe(wxChoice *liste)
281 {
282  wxString chemin = Parametre::Get()->getRepertoireParametre(_T("Liste_Couleur.txt"));
283  wxString repertoire = Parametre::Get()->getRepertoireParametre(_T("Preference"), _T("Couleur"));
284 
285  wxString copie;
286  wxDir Repertoire(repertoire);
287  wxFile fichier;
288  fichier.Create(chemin, true);
289  int nombre = 0;
290  bool continuer = Repertoire.GetFirst(&copie, _T("*.sauve"), wxDIR_FILES);
291 
292  while (continuer)
293  {
294  fichier.Write(copie);
295  fichier.Write(_T("\r\n"));
296  nombre++;
297  liste->Append(copie);
298  continuer = Repertoire.GetNext(&copie);
299  }
300  fichier.Close();
301  return nombre;
302 }
