82 void ImageText::Creer(wxString nom, int largeur, int largeurEcran, int hauteur)

96  Dc.SelectObject(wxNullBitmap);

98  *m_img = bitmap.ConvertToImage();
99 
100  m_largeur = m_img->GetWidth();

102

104

108  //int bytesPerPixel = 4;//m_img->HasAlpha() ? 4 : 3;

113  int rev_val=m_hauteur-1;
114  register int point, point2;
115 
116  for(int y=0; y<m_hauteur; y++)
117  {
