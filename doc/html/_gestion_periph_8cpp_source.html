165 void GestPeriph::ListePeripherique()
166 {
167  bool inclu;
168  ElementLecteurRacineFichier *t = NULL;
169  for (unsigned int i = 0; i<m_listeLecteur->GetCount(); i++)
170  {
171  t = m_listeLecteur->Item(i);
172  if (t->GetUtilise())
173  {
174  m_menuPeriph->Remove(t->GetMenuItem());
175  t->PasUtilise();
176  }
177  }
178 
179  wxArrayString chemin = wxFSVolume().GetVolumes();
180  unsigned int j = 0;
181 
182  for (size_t i = 0; i < chemin.Count(); i++)
183  {
184  if (wxDirExists(chemin[i]))
185  {
186  inclu = false;
187  j = 0;
188  while (j < m_listeLecteur->GetCount() && !inclu)
189  {
190  if (m_listeLecteur->Item(j)->GetChemin() == chemin[i])
191  {
192  t = m_listeLecteur->Item(j);
193  inclu = true;
194  }
195  j++;
196  }
197 
198  if (!inclu)
199  {
200  wxFSVolume volume(chemin[i]);
201  t = Insertion(m_listeLecteur, chemin[i], volume.GetDisplayName());
202  }
203 
204  t->SetMenuItem(m_menuPeriph->AppendSubMenu(t->GetMenu(), t->GetNomAffiche()));
205  t->SetUtilise(true);
206  }
207  }
208 }

265 ElementLecteurRacineFichier *GestPeriph::Insertion(ArrayOfElementLecteurRacineFichier *liste, wxString chemin, wxString type)
266 {
267  ElementLecteurRacineFichier *nouveau = new ElementLecteurRacineFichier();
268 
269  if (liste == NULL || nouveau == NULL)
270  return NULL;
271 
272  nouveau->SetChemin(chemin);
273  #ifdef __WINDOWS__
274  nouveau->SetNomAffiche(type);
275  #else
276  nouveau->SetNomAffiche(type + " (" + chemin.BeforeLast(wxFileName::GetPathSeparator()) + ")");
277  #endif
278 
279  nouveau->SetIdGauche(wxNewId());
280  Connect(nouveau->GetIdGauche(), wxEVT_COMMAND_MENU_SELECTED, wxCommandEventHandler(GestPeriph::AffListeRepertoireLect));
281  nouveau->SetIdDroite(wxNewId());
282  Connect(nouveau->GetIdDroite(), wxEVT_COMMAND_MENU_SELECTED, wxCommandEventHandler(GestPeriph::AffListeRepertoireLect));
283 
284  nouveau->CreerMenu();
285  nouveau->SetUtilise(false);
286  liste->Add(nouveau);
287  return nouveau;
288 }

416 void GestPeriph::ComparaisonStatut()
417 {
418  if (m_listeG->GetEtat() == VIDE && m_listeD->GetEtat() == VIDE)
419  {
420  for (int i=0; i<4; i++)
421  m_boutonModif[i].Enable(false);
422  }
423  else if (m_listeG->GetEtat() == VIDE || m_listeD->GetEtat() == VIDE)
424  {
425  for (int i=0; i<3; i++)
426  m_boutonModif[i].Enable(false);
427  m_boutonModif[3].Enable(true);
428  }
429  else if (m_listeG->GetEtat() == DOSSIER && m_listeD->GetEtat() == DOSSIER)
430  {
431  for (int i=0; i<4; i++)
432  m_boutonModif[i].Enable(true);
433  }
434  else if ((m_listeG->GetEtat() == M3U || m_listeG->GetEtat() == PLAYLIST) && (m_listeD->GetEtat() == M3U || m_listeD->GetEtat() == PLAYLIST))
435  {
436  for (int i=0; i<4; i++)
437  m_boutonModif[i].Enable(true);
438  }
439  else if (m_listeG->GetEtat() == DOSSIER && (m_listeD->GetEtat() == M3U || m_listeD->GetEtat() == PLAYLIST))
440  {
441  m_boutonModif[0].Enable(false);
442  m_boutonModif[1].Enable(false);
443  m_boutonModif[2].Enable(true);
444  m_boutonModif[3].Enable(true);
445  }
446  else if (m_listeD->GetEtat() == DOSSIER && (m_listeG->GetEtat() == M3U || m_listeG->GetEtat() == PLAYLIST))
447  {
448  m_boutonModif[0].Enable(true);
449  m_boutonModif[1].Enable(false);
450  m_boutonModif[2].Enable(false);
451  m_boutonModif[3].Enable(true);
452  }
453 }
