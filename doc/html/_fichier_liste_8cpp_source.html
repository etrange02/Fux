399 bool wxFichierListe::PlacerLigneInt(wxArrayString *ligne, long position, bool supprime)
400 {
401  wxTextFile fichierListe(m_Liste);
402  if (!fichierListe.Open())
403  return false;
404 
405  int j = 0, k = 0;
406  long val;
407  bool deplace = true;
408  wxArrayString chaine;
409  chaine.Alloc(ligne->GetCount());
410 
411  if (position < 0)
412  position = 0;
413  else if (position > long(fichierListe.GetLineCount()+1))
414  position = fichierListe.GetLineCount()+1;
415 
416  for (unsigned int i=0; i<ligne->GetCount(); i++)
417  {
418  ligne->Item(i).ToLong(&val);
419  chaine.Add(fichierListe.GetLine(val));
420  if (val == position)
421  deplace = false;
422  }
423  if (deplace)
424  {
425  for (unsigned int i=0; i<ligne->GetCount(); i++)
426  {
427  ligne->Item(i).ToLong(&val);
428  if (val > position)//Remonte
429  {
430  if (supprime)
431  fichierListe.RemoveLine(val);
432  fichierListe.InsertLine(chaine.Item(i), position+j);
433  j++;
434  }
435  else if (val < position)//Descend
436  {
437  if (supprime)
438  fichierListe.RemoveLine(val-k);
439  fichierListe.InsertLine(chaine.Item(i), position-1);
440  k++;
441  }
442  }
443  }
444  m_nombreFichier = fichierListe.GetLineCount();
445  fichierListe.Write();
446  fichierListe.Close();
447  return deplace;
