145 int PrefSon::CreerListe(wxChoice *liste)
146 {
147  wxString chemin = Parametre::Get()->getRepertoireParametre(_T("Liste_Son.txt"));
148 
149  wxString copie;
150  wxDir Repertoire(Parametre::Get()->getRepertoireParametre(_T("Preference"), _T("Son")));
151  wxFile fichier;
152  fichier.Create(chemin, true);
153  int nombre = 0;
154  bool continuer = Repertoire.GetFirst(&copie, _T("*.sauve"), wxDIR_FILES);
155 
156  while (continuer)
157  {
158  fichier.Write(copie);
159  fichier.Write(_T("\r\n"));
160  nombre++;
161  liste->Append(copie);
162  continuer = Repertoire.GetNext(&copie);
163  }
164  fichier.Close();
165  return nombre;
166 }

248 void PrefSon::Son_Bouton_Enregistrer(wxCommandEvent &WXUNUSED(event))
249 {
250  if (!m_boiteNomMod->IsEmpty() && (m_listeModif_Nouveau == NOUVEAU || m_listeModif->GetCurrentSelection() != wxNOT_FOUND))
251  {
252  wxFile FichierListe;
253  wxString chemin = Parametre::Get()->getRepertoireParametre(_T("Preference"), _T("Son"), wxFileName::GetPathSeparator()), cheminFichier, nom = m_boiteNomMod->GetValue(), volume, volumePC;
254 
255  cheminFichier = chemin;
256  chemin << nom << _T(".sauve");
257 
258  volume << m_boiteVolMod->GetValue();
259  volumePC << m_boiteVolPCMod->GetValue();
260 
261  if (wxFileName::FileExists(chemin) && m_listeModif_Nouveau == NOUVEAU && !m_Bouton_Appliquer->IsEnabled())
262  {
263  wxMessageBox(_("Un fichier portant ce nom existe déjà !"), _("Erreur !"));
264  }
265  else
266  {
267  if (m_listeModif_Nouveau == MODIFIER)// && m_boiteNomMod->IsModified())// || (m_listeModif_Nouveau == NOUVEAU && m_Bouton_Appliquer->IsEnabled())
268  {
269  cheminFichier << m_listeModif->GetString(m_listeModif->GetCurrentSelection());
270  wxRemoveFile(cheminFichier);
271  }
272  if (!FichierListe.Create(chemin, true))
273  {
274  wxMessageBox(_("Erreur dans le nom.\nVérifiez que vous utilisez des caractères autorisés."), _("Erreur"));
275  return;
276  }
277  FichierListe.Write(_T("#EXTSAUVE_S1\r\n"));
278  FichierListe.Write(nom + _T("\r\n"));
279  //FichierListe.Write();
280  FichierListe.Write(volume + _T("\r\n"));
281  //FichierListe.Write("\r\n");
282  FichierListe.Write(volumePC);
283  FichierListe.Write(_T("\r\n0.5\r\n0.5"));
284  FichierListe.Close();
285  m_listeModif->Clear();
286  CreerListe(m_listeModif);
287  if (m_listeModif_Nouveau == MODIFIER)
288  m_listeModif->SetStringSelection(nom + _T(".sauve"));
289  else
290  m_Bouton_Appliquer->Enable(true);
291  }
292  }
293  else
294  wxMessageBox(_("Vous devez renseigner tous les champs"), _("Erreur !"));
295 }
