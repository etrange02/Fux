261 Couleur MusiqueGraph::CouleurExtrem(int hauteur, int sizer_h, e_GraphPosition niveau)

267  r = m_haut.R;
268  else if (m_haut.R > m_miSup.R)
269  r = 255*hauteur/(sizer_h-19);
270  else
271  r = 255-255*hauteur/(sizer_h-19);
272 
273  if (m_haut.V == m_miSup.V)
274  v = m_haut.V;
275  else if (m_haut.V > m_miSup.V)
276  v = 255*hauteur/(sizer_h-19);
277  else
278  v = 255-255*hauteur/(sizer_h-19);
279 
280  if (m_haut.B == m_miSup.B)
281  b = m_haut.B;
282  else if (m_haut.B > m_miSup.B)
283  b = 255*hauteur/(sizer_h-19);
284  else
285  b = 255-255*hauteur/(sizer_h-19);
286  }
287  else if (niveau == BAS_INF)
288  {
289  if (m_miInf.R == m_bas.R)
290  r = m_miInf.R;
291  else if (m_miInf.R > m_bas.R)
292  r = 255-255*hauteur/(sizer_h-19);
293  else
294  r = 255*hauteur/(sizer_h-19);
295 
296  if (m_miInf.V == m_bas.V)
297  v = m_miInf.V;
298  else if (m_miInf.V > m_bas.V)
299  v = 255-255*hauteur/(sizer_h-19);
300  else

302

306  b = 255-255*hauteur/(sizer_h-19);

388 void MusiqueGraph::OnToucheDown(wxKeyEvent &event)
389 {
390  if (event.GetKeyCode() == 'P')
391  {
392  if (Musique::Get()->GetLecture())
393  Musique::Get()->SetPause(true);
394  else if (Musique::Get()->GetPause())
395  Musique::Get()->SetPause(false);
396  else
397  Musique::Get()->ChangementChanson(IDENTIQUE);
398  }
399  else if (event.GetKeyCode() == 'S')
400  Musique::Get()->SetStop();
401  else if (event.GetKeyCode() == 'A')
402  Musique::Get()->SetAleatoire(!Musique::Get()->GetAleatoire());
403  else if (event.GetKeyCode() == '+' || event.GetKeyCode() == WXK_ADD || event.GetKeyCode() == WXK_NUMPAD_ADD)
404  SliderSon::Get()->SonUp();
405  else if (event.GetKeyCode() == '-' || event.GetKeyCode() == WXK_SUBTRACT || event.GetKeyCode() == WXK_NUMPAD_SUBTRACT || event.GetKeyCode() == '6')
406  SliderSon::Get()->SonDown();
407  else if (event.GetKeyCode() == WXK_RIGHT)
408  Musique::Get()->ChangementChanson(SUIVANT);
409  else if (event.GetKeyCode() == WXK_LEFT)
410  Musique::Get()->ChangementChanson(PRECEDENT);
411  else if (event.GetKeyCode() == WXK_DELETE)
412  {
413  wxCommandEvent evt(wxEVT_MUSIQUE_SUPPRESSION, GetId());
414  GetParent()->GetEventHandler()->AddPendingEvent(evt);
415  }
416  else
417  event.Skip();
418 }
