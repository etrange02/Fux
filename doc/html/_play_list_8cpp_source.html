141 void PlayList::EnregistrerM3U(wxCommandEvent &WXUNUSED(event))
142 {
143  DialogEnregistreM3U fen(this, wxID_ANY, wxEmptyString);
144  fen.Creer();
145  bool modif = true;
146 
147  if (fen.ShowModal() == wxID_OK)
148  {
149  wxTextFile fichierLec(Parametre::Get()->getRepertoireParametre(_T("musique.liste")));
150  if (!fichierLec.Open())
151  return;
152  if (fen.GetName().IsEmpty())
153  return;
154  wxTextFile fichierEcr(fen.GetChemin());
155 
156  if (fichierEcr.Exists())
157  {
158  wxMessageDialog boite2(NULL, _("Fichier déjà existant. Souhaitez-vous le remplacer ?"), _("Fichier déjà existant"), wxYES_NO|wxICON_QUESTION|wxCENTRE|wxYES_DEFAULT);
159  if (boite2.ShowModal() != wxID_YES)
160  modif = false;
161  else
162  {
163  fichierEcr.Open();
164  fichierEcr.Clear();
165  }
166  }
167  else
168  {
169  if (!fichierEcr.Create())
170  {
171  wxLogError(_("Erreur dans le nom.\nVérifiez que vous utilisez des caractères autorisés."), _("Erreur"));
172  return;
173  }
174  }
175  if (modif)
176  {
177  fichierEcr.AddLine(_T("#EXTM3U"));
178  for (unsigned int i=0; i<fichierLec.GetLineCount(); i++)
179  fichierEcr.AddLine(fichierLec.GetLine(i));
180  fichierEcr.Write();
181 
182  if (fen.GetCheminRaccourci() != wxEmptyString && !wxFileExists(fen.GetCheminRaccourci()))
183  {
184  if (!CreationRaccourci(fen.GetCheminRaccourci(), fen.GetChemin()))
185  {
186  wxLogMessage(_("Echec de la création du raccourci."));
187  }
188  }
189  }
190  fichierLec.Close();
191  fichierEcr.Close();
192  }
193  fen.Destroy();
194 }

249 void PlayList::OnAppliquerTAG(wxCommandEvent &WXUNUSED(event))
250 {

264  m_ObjetTAG.tag()->setYear(m_BoiteAnnee->GetValue());//Année

266  m_ObjetTAG.tag()->setGenre("");//Genre
267  else
268  m_ObjetTAG.tag()->setGenre(TagLib::String(m_BoiteGenre->GetValue().fn_str()));//Genre
269  m_ObjetTAG.save();
270  m_ObjetTAG = TagLib::FileRef(_T(""));
271 
272  int i = tempo.Find(wxFileName::GetPathSeparator(), true);
273  tempo = tempo.Left(i);
274  tempo << wxFileName::GetPathSeparator() << m_BoiteNom->GetValue();//Nom du fichier
275  if (fichierTAG != tempo)
276  {
277  if (wxRenameFile(fichierTAG, tempo, true))
278  m_musique->GetFichier()->EchangeNom(fichierTAG, tempo);

344 void PlayList::EvtImage(wxCommandEvent &event)

356  TagLib::MPEG::File f(TagLib::FileName(fichierTAG.fn_str()));

387  }
