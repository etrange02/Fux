91 void* ThreadFichierFichier::Entry()
92 {
93  m_estActif = true;
94  m_continue = true;
95  ElementThreadFichier *element = NULL;
96  wxString chemin;
97 
98  while (m_continue && !TestDestroy())
99  {
100  if (m_liste->IsEmpty())
101  Sleep(500);
102  else
103  {
104  element = m_liste->Item(0);
105 
106  if (element->IsAction(COPIE))
107  {
108  chemin = element->GetDestination() + wxFileName::GetPathSeparator() + element->GetNom().AfterLast(wxFileName::GetPathSeparator());
109 
110  if (wxFileExists(element->GetNom()))
111  wxCopyFile(element->GetNom(), chemin, false);
112  else if (wxDirExists(element->GetNom()))
113  {
114  DossierCopie(element->GetNom(), chemin);
115  FichierCopie(element->GetNom(), chemin);
116  }
117  }
118  else if (element->IsAction(DEPLACE))
119  {
120  chemin = element->GetDestination() + wxFileName::GetPathSeparator() + element->GetNom().AfterLast(wxFileName::GetPathSeparator());
121 
122  if (wxFileExists(element->GetNom()))
123  wxRenameFile(element->GetNom(), chemin, false);
124  else if (wxDirExists(element->GetNom()))
125  {
126  DossierCopie(element->GetNom(), chemin);
127  FichierDeplace(element->GetNom(), chemin);
128  }
129  }
130  else if (element->IsAction(SUPPRIME))
131  {
132  if (wxFileExists(element->GetNom()))
133  wxRemoveFile(element->GetNom());
134  else if (wxDirExists(element->GetNom()))
135  FichierSuppression(element->GetNom());
136  }
137 
138  element->~ElementThreadFichier();
139  m_liste->Remove(element);
140  }
141  }
142  m_estActif = false;
143  m_continue = false;
144  return NULL;
145 }
