79 void wxFuXApp::EnvoiStringAutreInstance()
80 {
81  TCPClient *client = new TCPClient();
82  TCPConnexionEchangeInstanceLocalHost *connexion = (TCPConnexionEchangeInstanceLocalHost*) client->MakeConnection(IPC_HOST, IPC_SERVICE, IPC_TOPIC);
83  wxFileName chaine;
84 
85  if (connexion)
86  {
87  for (int i = 1; i<argc; i++)
88  {
89  chaine = argv[i];
90  chaine.Normalize(wxPATH_NORM_SHORTCUT);
91  if (Parametre::Get()->islisable(chaine.GetFullPath().AfterLast('.').Lower()))
92  {
93  if (!connexion->Execute(chaine.GetFullPath()))
94  {
95  wxLogMessage(_T("Erreur lors de l'envoie du titre à l'application principale : \n") + chaine.GetFullPath());
96  }
97  }
98  else
99  wxLogMessage(_T("L'extension ") + chaine.GetExt() + _T(" n'est pas prise en compte"));
100  }
101  connexion->Disconnect();
102  delete connexion;
103  }
104  else
105  wxLogMessage(_T("Impossible d'envoyer le titre à l'application principale\n") + chaine.GetFullPath());
106  delete client;
107 }
