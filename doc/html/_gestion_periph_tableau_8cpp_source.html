126 bool ListeGestionPeriph::AfficheContenuDossier(wxString chemin, bool cache, bool filtre)
127 {

129  return false;

131  bool cont;
132  wxString temp;

134  wxDir repertoire(chemin);

136  DeleteAllItems();
137 //Dossiers
138  cont = (cache) ? repertoire.GetFirst(&temp, wxEmptyString, wxDIR_DIRS | wxDIR_HIDDEN) : repertoire.GetFirst(&temp, wxEmptyString, wxDIR_DIRS);
139 
140  while (cont)
141  {
142  j = temp.Find(wxFileName::GetPathSeparator(), true);
143  pos = GetItemCount();
144  pos = InsertItem(pos, temp.Right(temp.Length()-j-1));
145  SetItem(pos, 1, _T("Dossier"));

153  if (filtre)
154  {
155  while (cont)
156  {
157  if (Parametre::Get()->islisable(temp.AfterLast('.').Lower()) || temp.Lower().EndsWith(_T(".m3u")))// || temp.Lower().EndsWith(".m4a")/*|| temp.Lower().EndsWith(".wav") ||*/ /*temp.Lower().EndsWith(".aac") || temp.Lower().EndsWith(".ogg")*/)

159  j = temp.Find(wxFileName::GetPathSeparator(), true);
160  pos = GetItemCount();
161  pos = InsertItem(pos, temp.Right(temp.Length()-j-1));
162  SetItem(pos, 1, temp.AfterLast('.'));
163  }
164  cont = repertoire.GetNext(&temp);
165  }
166  }
167  else
168  {
169  while (cont)
170  {
171  j = temp.Find(wxFileName::GetPathSeparator(), true);
172  pos = GetItemCount();
173  pos = InsertItem(pos, temp.Right(temp.Length()-j-1));
174  SetItem(pos, 1, temp.AfterLast('.'));
175  cont = repertoire.GetNext(&temp);
176  }
177  }
178 
179  SetFocus();
180  //if (GetItemCount() != 0)
